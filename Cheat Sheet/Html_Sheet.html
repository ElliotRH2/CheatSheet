const TweenService = game.GetService("TweenService");

const part = new Instance("Part");
part.Position = new Vector3(0, 10, 0);
part.Color = new Color3(1, 0, 0);
part.Anchored = true;
part.Parent = game.Workspace;
         
const goal = {};
goal.Position = new Vector3(10, 10, 0);
goal.Color = new Color3(0, 1, 0);

const tweenInfo = new TweenInfo(5);
         
const tween = TweenService.Create(part, tweenInfo, goal);
         
tween.Play();
